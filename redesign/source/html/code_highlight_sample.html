MooEditable.Actions.h2 = {
	title:'H2',
	type:'button',
	options:{
		mode:'icon'
	},
	command: function(name){
		var argument = '<' + name.name + '>';
		this.focus();
		this.execute('formatBlock', false, argument);
	}
};
